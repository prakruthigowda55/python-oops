class Student:
    def __init__(self,phy,chem,math):
        self.phy=phy
        self.chem=chem
        self.math=math

    @property
    def percentage(self):
        return str((self.phy + self.chem + self.math) / 3) + "%"

std1=Student(98,99,100)
print(std1.percentage)

std1.phy=89
print(std1.phy)
print(std1.percentage)
